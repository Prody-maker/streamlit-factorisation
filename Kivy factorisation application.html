from kivy.uix.boxlayout import BoxLayout
from kivy.uix.label import Label
from kivy.uix.textinput import TextInput
from kivy.uix.button import Button
from sympy import symbols, factor, Add, collect
from sympy.parsing.sympy_parser import parse_expr, standard_transformations, implicit_multiplication_application

# Dfinir les symboles autoriss
x, y, a, b = symbols('x y a b')
transformations = standard_transformations + (implicit_multiplication_application,)

class FactorisationLayout(BoxLayout):
    def __init__(self, **kwargs):
        super().__init__(orientation='vertical', spacing=10, padding=20, **kwargs)

        self.label = Label(text=" Entrez une expression (ex: 7x + 3x):", size_hint=(1, 0.2))
        self.input = TextInput(multiline=False, hint_text="7x + 3x", size_hint=(1, 0.2))

        self.button = Button(text="Factoriser", size_hint=(1, 0.3))

        self.button.bind(on_press=self.factoriser)

        self.add_widget(self.label)
        self.add_widget(self.input)
        self.add_widget(self.button)
        self.add_widget(self.result)

    def factoriser(self, instance):
        expr_str = self.input.text.replace('^', '**')
        try:

from kivy.app import App
from kivy.uix.boxlayout import BoxLayout
from kivy.uix.label import Label
from kivy.uix.textinput import TextInput
from kivy.uix.button import Button
from sympy import symbols, factor, collect, Add
from sympy.parsing.sympy_parser import parse_expr, standard_transformations, implicit_multiplication_application

# Dfinir les symboles autoriss
x, y, a, b = symbols('x y a b')
transformations = standard_transformations + (implicit_multiplication_application,)

class FactorisationLayout(BoxLayout):
    def __init__(self, **kwargs):
        super().__init__(orientation='vertical', spacing=10, padding=20, **kwargs)

        self.label = Label(text=" Entrez une expression (ex: 7x + 3x):", size_hint=(1, 0.2))
        self.input = TextInput(multiline=False, hint_text="7x + 3x", size_hint=(1, 0.2))
        self.result = Label(text="", size_hint=(1, 0.3))
        self.button = Button(text="Factoriser", size_hint=(1, 0.3))

        self.button.bind(on_press=self.factoriser)

        self.add_widget(self.label)
        self.add_widget(self.input)
        self.add_widget(self.button)
        self.add_widget(self.result)

    def factoriser(self, instance):
        expr_str = self.input.text.replace('^', '**')
        try:
            # Ne pas valuer automatiquement
            expression = parse_expr(expr_str, transformations=transformations, evaluate=False)

            # Obtenir les termes de l'expression si c'est une somme
            if isinstance(expression, Add):
                # Rechercher les facteurs communs entre tous les termes
                facteurs_communs = expression.as_coefficients_dict()

                # Vrifier sil y a un facteur commun (comme x)
                facteurs = list(facteurs_communs.keys())
                variables = set()
                for terme in facteurs:
                    if terme.is_Mul:
                        for f in terme.args:
                            if f.is_Symbol:
                                variables.add(f)
                    elif terme.is_Symbol:
                        variables.add(terme)

                if len(variables) == 1:
                    var = list(variables)[0]
                    coeffs = []
                    for terme in expression.args:
                        coeff = terme / var
                        coeffs.append(coeff)

                    nouvelle_expr = f"{var}(" + ' + '.join([str(c) for c in coeffs]) + ")"
                    affichage_propre = nouvelle_expr.replace('**', '^').replace('*', '')
                    self.result.text = f"Rsultat : {affichage_propre}"
                    return

            # Sinon, utiliser la factorisation normale
            expression_factoree = factor(expression)
            affichage_propre = str(expression_factoree).replace('**', '^').replace('*', '')
            self.result.text = f"Rsultat : {affichage_propre}"

        except Exception as e:
            self.result.text = f"Erreur : {e}"

class FactorisationApp(App):
    def build(self):
        return FactorisationLayout()

if __name__ == '__main__':
    FactorisationApp().run()

import streamlit as st
from sympy import symbols, Add, factor
from sympy.parsing.sympy_parser import (
    parse_expr,
    standard_transformations,
    implicit_multiplication_application,
)

# Dfinir les symboles
x, y, a, b = symbols("x y a b")
transformations = standard_transformations + (implicit_multiplication_application,)

# Titre de la page
st.title(" Factorisation d'expressions algbriques")

# Zone de saisie utilisateur
expr_str = st.text_input("Entrez une expression (ex: 7x + 3x)", "7x + 3x")

def factoriser_personnalise(expression):
    # Analyse des termes si c'est une somme
    if isinstance(expression, Add):
        facteurs = expression.as_coefficients_dict()
        variables = set()

        for terme in facteurs.keys():
            if terme.is_Mul:
                for t in terme.args:
                    if t.is_Symbol:
                        variables.add(t)
            elif terme.is_Symbol:
                variables.add(terme)

            for terme in expression.args:

    # Sinon factorisation normale
    return str(factor(expression)).replace("**", "^").replace("*", "")

if expr_str:
    try:
        expr = parse_expr(expr_str.replace("^", "**"), transformations=transformations, evaluate=False)
        resultat = factoriser_personnalise(expr)
        st.success(f"Rsultat : {resultat}")
    except Exception as e:
